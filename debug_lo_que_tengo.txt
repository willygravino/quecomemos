    51	  async function postFormData({ url, csrf, fd }) {
    52	    const res = await fetch(url, {
    53	      method: "POST",
    54	      headers: {
    55	        "X-Requested-With": "XMLHttpRequest",
    56	        "X-CSRFToken": csrf,
    57	      },
    58	      body: fd,
    59	      credentials: "same-origin",
    60	    });
    61	
    62	    const ct = res.headers.get("content-type") || "";
    63	    if (!ct.includes("application/json")) {
    64	      const text = await res.text();
    65	      console.error("❌ Respuesta no JSON:", res.status, text.slice(0, 300));
    66	      return;
    67	    }
    68	
    69	    const data = await res.json().catch(() => ({}));
    70	    if (!res.ok || data.success === false) {
    71	      console.error("❌ POST error:", res.status, data);
    72	    }
    73	  }
